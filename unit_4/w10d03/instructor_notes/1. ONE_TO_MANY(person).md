# One to Many Relationship in Rails

## Add a Foreign Key Column

in `psql`:

```
ALTER TABLE people ADD COLUMN home_id INT;
```

in `app/models/person.rb`:

```ruby
attr_reader :id, :name, :age, :home_id #add attribute reader for home_id

DB = PG.connect(host: "localhost", port: 5432, dbname: 'contacts')

def initialize(opts)
    @id = opts["id"].to_i
    @name = opts["name"]
    @age = opts["age"].to_i
    if opts["home_id"] #if home_id is in opts hash, show it
        @home_id = opts["home_id"].to_i
    end
end
#...
def self.create(opts)
    #when creating, use home_id if it exists, "NULL", otherwise
    results = DB.exec(
        <<-SQL
            INSERT INTO people (name, age, home_id)
            VALUES ( '#{opts["name"]}', #{opts["age"]},  #{opts["home_id"] ? opts["home_id"] : "NULL"})
            RETURNING id, name, age, home_id;
        SQL
    )
    return Person.new(results.first)
end
#...
def self.update(id, opts)
    results = DB.exec(
        # add home_id to update
        <<-SQL
            UPDATE people
            SET name='#{opts["name"]}', age=#{opts["age"]}, home_id=#{opts["home_id"] ? opts["home_id"] : "NULL"}
            WHERE id=#{id}
            RETURNING id, name, age, home_id;
        SQL
        # make sure to return home_id
    )
    return Person.new(results.first)
end
```

## Expand a foreign key to an object

First convert a Person's `home_id` to a `home` which is a Location

In `app/models/person.rb`:

```ruby
class Person
    attr_reader :id, :name, :age, :home # add home attr_reader
    DB = PG.connect(host: "localhost", port: 5432, dbname: 'contacts')

    def initialize(opts)
        @id = opts["id"].to_i
        @name = opts["name"]
        @age = opts["age"].to_i
        if opts["home"] # change home_id to home
            @home = opts["home"]
        end
    end

    def self.all
        results = DB.exec(
            <<-SQL
                SELECT
                    people.*,
                    locations.street,
                    locations.city,
                    locations.state
                FROM people
                LEFT JOIN locations
                    ON people.home_id = locations.id
            SQL
        )
        return results.map do |result|
            if result["home_id"]
                home = Location.new(
                    {
                        "id" => result["home_id"],
                        "street" => result["street"],
                        "city" => result["city"],
                        "state" => result["state"],
                    }
                )
            end
            Person.new(
                {
                    "id" => result["id"],
                    "name" => result["name"],
                    "age" => result["age"],
                    "home" => home,
                }
            )
        end
    end

    def self.find(id)
        results = DB.exec(
            <<-SQL
                SELECT
                    people.*,
                    locations.street,
                    locations.city,
                    locations.state
                FROM people
                LEFT JOIN locations
                    ON people.home_id = locations.id
                WHERE people.id=#{id};
            SQL
        )
        result = results.first
        if result["home_id"]
            home = Location.new(
                {
                    "id" => result["home_id"],
                    "street" => result["street"],
                    "city" => result["city"],
                    "state" => result["state"],
                }
            )
        end
        return Person.new(
            {
                "id" => result["id"],
                "name" => result["name"],
                "age" => result["age"],
                "home" => home,
            }
        )
    end
    #...
end
```
